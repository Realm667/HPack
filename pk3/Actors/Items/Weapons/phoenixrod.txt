// Phoenix Rod -------------------------------------------------------

ACTOR HPPhoenix : PhoenixRod replaces PhoenixRod
{
	//$Category Weapons
	//$Title Phoenix Rod
	+WEAPON.NOAUTOFIRE
	Tag "$TAG_PHOENIXROD"
	Weapon.SelectionOrder 2500
	Weapon.Sisterweapon "HPPhoenixPowered"
	Obituary "$OB_MPPHOENIXROD"
	States
    {
    Fire:
    	PHNX B 5 BRIGHT A_PlaySoundEx("weapons/phoenixshoot2","Auto",0)
    	PHNX C 0 BRIGHT A_Recoil(4)
    	PHNX C 7 BRIGHT A_FireCustomMissile("HPPhoenixFX1", 0, 1, 0, 0) //A_FirePhoenixPL1
    	PHNX DB 4 BRIGHT
    	PHNX B 0 BRIGHT A_ReFire
    	Goto Ready
	}
}

ACTOR HPPhoenixPowered : PhoenixRodPowered
{
	+WEAPON.POWERED_UP
	Tag "$TAG_PHOENIXROD"
	Obituary "$OB_MPPHOENIXROD"
	Weapon.SelectionOrder 2500
	Weapon.SisterWeapon "HPPhoenix"
	Weapon.AmmoGive 0
	States
	{
	Ready:
		PHNX AAAAEEEEFFFFEEEE 1 A_WeaponReady
		Loop
	Fire:
		PHNX B 3 BRIGHT Offset(0,36) A_InitPhoenixPL2
	Hold:
		PHNX C 1 BRIGHT Offset(0,36) A_FirePhoenixPL2
		PHNX B 4 BRIGHT Offset(0,40) A_ReFire
	Powerdown:
		PHNX A 4 BRIGHT Offset(0,36) A_ShutdownPhoenixPL2
		Goto Ready
	}
}

ACTOR HPPhoenixFX1 : PhoenixFX1
{
  Scale 0.5
  States
  {
  Spawn:
    PHOX A 4 Bright A_PhoenixPuff
    Loop
  Death:
    FX08 A 0 Bright A_SetScale(1.0)
    FX08 A 6 Bright A_Explode
    FX08 B 5 Bright
    FX08 C 5 Bright
    FX08 D 4 Bright
    FX08 E 4 Bright
    FX08 FGH 4 Bright
    Stop
  }
}
