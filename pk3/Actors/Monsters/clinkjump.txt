actor JumpClink : HPClink 19273
{
  //$Category Monsters
  //$Title Sabreclaw (Jumping)
  CameraHeight 49
  +SHORTMISSILERANGE
  +NOTARGETSWITCH
  +DONTHURTSPECIES
  +MISSILEMORE
  States
  {
  See:
    CLNK A 0 A_Chase("Melee", "")
    CLNK BCD 3 A_Chase
    Loop
  Missile:
    CLNK P 0 A_Jump(16,"Jump")
    Goto See
  Jump:
    CLNK P 6
    CLNK Q 1 ThrustThingZ (0, 40, 0, 1)
    CLNK Q 15 A_Recoil (-10)
    CLNK P 6
    Goto See
  Jump2: //Use SetActorState for these three states
    CLNK P 6 A_ChangeFlag("NoPain", 1)
    CLNK Q 1 ThrustThingZ (0, 40, 0, 1)
    CLNK Q 15 A_Recoil (-10)
    CLNK P 6 A_ChangeFlag("NoPain", 0)
    Goto See
  JumpDown: //ACS-activated
    CLNK P 0 A_ChangeFlag("NoPain", 1)
    CLNK P 8
    CLNK Q 8 ThrustThing(Angle*256/360, 20, 0, 0)
    CLNK Q 5 A_Stop
  Falling: //ACS-activated
    CLNK Q 0 A_ChangeFlag("NoPain", 1)
    CLNK Q 1 A_CheckFloor(1)
    Loop
    CLNK Q 2
    CLNK P 12
    CLNK P 0 A_ChangeFlag("NoPain", 0)
    Goto Spawn
  }
}
